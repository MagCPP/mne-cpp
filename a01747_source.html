88 MatrixXd RtFilter::filterChannelsConcurrently(const MatrixXd& matDataIn,
89  int iMaxFilterLength,
90  const QVector<int>& lFilterChannelList,
91  const QList<FilterData>& lFilterData)
92 {
93  //Initialise the overlay matrix
94  if(m_matOverlap.cols() != iMaxFilterLength || m_matOverlap.rows() < matDataIn.rows()) {
95  m_matOverlap.resize(matDataIn.rows(), iMaxFilterLength);
96  m_matOverlap.setZero();
97  }
98 
99  if(m_matDelay.cols() != iMaxFilterLength/2 || m_matOverlap.rows() < matDataIn.rows()) {
100  m_matDelay.resize(matDataIn.rows(), iMaxFilterLength/2);
101  m_matDelay.setZero();
102  }
103 
104  //Resize output matrix to match input matrix
105  MatrixXd matDataOut = matDataIn;
106 
107  //Generate QList structure which can be handled by the QConcurrent framework
108  QList<QPair<QList<FilterData>,QPair<int,RowVectorXd> > > timeData;
109  QList<int> notFilterChannelIndex;
110 
111  //Only select channels specified in lFilterChannelList
112  for(qint32 i = 0; i < matDataIn.rows(); ++i) {
113  int pos = lFilterChannelList.indexOf(i);
114  if(pos != -1 && pos < matDataIn.rows()) {
115  timeData.append(QPair<QList<FilterData>,QPair<int,RowVectorXd> >(lFilterData,QPair<int,RowVectorXd>(pos,matDataIn.row(pos))));
116  } else {
117  notFilterChannelIndex.append(i);
118  }
119  }
120 
121  //Do the concurrent filtering
122  if(!timeData.isEmpty()) {
123  QFuture<void> future = QtConcurrent::map(timeData,
124  doFilterPerChannelRTMSA);
125 
126  future.waitForFinished();
127 
128  //Do the overlap add method and store in matDataOut
129  int iFilteredNumberCols = timeData.at(0).second.second.cols();
130 
131  for(int r = 0; r < timeData.size(); r++) {
132  //Get the currently filtered data. This data has a delay of filterLength/2 in front and back.
133  RowVectorXd tempData = timeData.at(r).second.second;
134 
135  //Perform the actual overlap add by adding the last filterlength data to the newly filtered one
136  tempData.head(iMaxFilterLength) += m_matOverlap.row(timeData.at(r).second.first);
137 
138  //Write the newly calulated filtered data to the filter data matrix. Keep in mind that the current block also effect last part of the last block (begin at dataIndex-iFilterDelay).
139  int start = 0;
140  matDataOut.row(timeData.at(r).second.first).segment(start,iFilteredNumberCols-iMaxFilterLength) = tempData.head(iFilteredNumberCols-iMaxFilterLength);
141 
142  //Refresh the m_matOverlap with the new calculated filtered data.
143  m_matOverlap.row(timeData.at(r).second.first) = timeData.at(r).second.second.tail(iMaxFilterLength);
144  }
145  }
146 
147  //Fill filtered data with raw data if the channel was not filtered
148  for(int i = 0; i < notFilterChannelIndex.size(); ++i) {
149  matDataOut.row(notFilterChannelIndex.at(i)) << m_matDelay.row(notFilterChannelIndex.at(i)), matDataIn.row(notFilterChannelIndex.at(i)).head(matDataIn.cols() - iMaxFilterLength/2);
150 
151  //matDataOut.row(notFilterChannelIndex.at(i)).segment(0, matDataIn.row(notFilterChannelIndex.at(i)).cols()) = matDataIn.row(notFilterChannelIndex.at(i));
152  }
153 
154  m_matDelay = matDataIn.block(0, matDataIn.cols()-iMaxFilterLength/2, matDataIn.rows(), iMaxFilterLength/2);
155 
156  return matDataOut;
157 }
